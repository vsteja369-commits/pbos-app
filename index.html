<script>

let currentTraits = [];

 

function addTrait(){

  let name = document.getElementById("traitName").value;

  let value = document.getElementById("traitValue").value;

  let unit = document.getElementById("traitUnit").value;

 

  if(name && value){

    currentTraits.push({name, value, unit});

    renderTraits();

  }

 

  document.getElementById("traitName").value="";

  document.getElementById("traitValue").value="";

  document.getElementById("traitUnit").value="";

}

 

function renderTraits(){

  let html="";

  currentTraits.forEach((t,i)=>{

    html += `<div class="small">${t.name}: ${t.value} ${t.unit}</div>`;

  });

  document.getElementById("traitsList").innerHTML = html;

}

 

function saveEntry(){

  let id = document.getElementById("lineId").value;

  let type = document.getElementById("lineType").value;

 

  if(!id) return alert("Enter Line ID");

 

  let entry = {

    id, type,

    traits: currentTraits,

    date: new Date().toLocaleString()

  };

 

  let data = JSON.parse(localStorage.getItem("pbos") || "[]");

  data.push(entry);

  localStorage.setItem("pbos", JSON.stringify(data));

 

  currentTraits = [];

  renderTraits();

  loadEntries();

}

 

function loadEntries(){

  let data = JSON.parse(localStorage.getItem("pbos") || "[]");

  let html="";

 

  data.forEach(e=>{

    html += `<div class="card">

      <b>${e.id}</b> (${e.type})<br>

      ${e.traits.map(t=>`${t.name}: ${t.value} ${t.unit}`).join("<br>")}

    </div>`;

  });

 

  document.getElementById("entries").innerHTML = html;

}

 

function exportExcel(){

  let data = JSON.parse(localStorage.getItem("pbos") || "[]");

 

  let rows = [];

 

  data.forEach(e=>{

    e.traits.forEach(t=>{

      rows.push({

        LineID: e.id,

        Type: e.type,

        Trait: t.name,

        Value: t.value,

        Unit: t.unit

      });

    });

  });

 

  let csv = "LineID,Type,Trait,Value,Unit\n";

  rows.forEach(r=>{

    csv += `${r.LineID},${r.Type},${r.Trait},${r.Value},${r.Unit}\n`;

  });

 

  let blob = new Blob([csv], {type:"text/csv"});

  let a = document.createElement("a");

  a.href = URL.createObjectURL(blob);

  a.download = "pbos_data.csv";

  a.click();

}

 

loadEntries();

</script>
